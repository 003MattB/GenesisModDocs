StepSorter.pm

Sorts steps in the order defined on the checklist. IE orig < pcb < orig_et < pcb_etch < cip[n] < creg[n] < lreg[n] < cim[n] < [ipc_]s_coup < [ipc_]d_coup < peel_coup < mil[*] = [ipc_]cl3 < other

SYNOPSIS

     ##################                          ##################
     ################## EXAMPLES OF SIMPLE USAGE ##################
     ##################                          ##################
     use StepSorter;
     use Matrix;
     use Genesis;
     my $f = new Genesis;
     my $matrix = Matrix->new($f, $ENV{JOB});
     
     my @steps = $matrix->getSteps();
     
     my $sorter = $StepSorter->new();
     
     my @sorted_steps = $sorter->sort(@steps);
     
     print "original ordering of steps in matrix\n";
     foreach my $step (@steps) {
          my $name = $step->getName();
          print "$name, ";
     }
     print "\n\nsorted order of steps\n";
     foreach my $step (@sorted_steps) {
          my $name = $step->getName();
          print "$name, ";
     }
     print "\n\n";
