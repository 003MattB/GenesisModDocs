SYNOPSIS

     ##################                          ##################
     ################## EXAMPLES OF SIMPLE USAGE ##################
     ##################                          ##################
     use DrillLayerSorter;
     
     my $job = Job->new(PTX => $ENV{JOB}, GENESIS => $f);
     my $sorter = DrillLayerSorter->new(JOB_MOD => $job);
     
     my @drill_layers = ("ldi","p11","p21","p31","p32","d2");
     
     #the default sort orders the layers by their suffix ie
     # d1 comes before d1, p11 comes before p21 followed by p22
     # layers without a numeric suffix will be put last in the sorting order
     my @sorted = $sorter->sort(@drill_layers);
     #result
     #("d2","p11","p21","p31","p31","ldi");
     
     #orders the drill layers by their drill start
     #so if p11 starts on layer 1a and p21 starts on layer 6a
     #p11 will come before p21
     my @sorted_by_start = $sorter->sortBy("ByDrillStart",@drill_layers);
     #result
     #("d2","p32","p31","ldi","p11","p21")
     
     #orders the drill layers by their drill end
     #so if p11 ends on layer 5a and p21 starts on layer 10a
     #p11 will come before p21
     my @sorted_by_end = $sorter->sortBy("ByDrillEnd",@drill_layers);
     #result
     #("p11","p21","d2","p32","p31","ldi");
